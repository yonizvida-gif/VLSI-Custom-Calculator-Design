// 4-bit ALU with 64-bit SRAM and Internal Address Counter
// Flow: Load First Num -> Load Second Num -> Auto-Store Result

module calculator (
    input clk,
    input reset,
    input signed [3:0] data_in,   // Data from user
    input load_en,         // Trigger to load data and increment counter
    input [1:0] op_sel,    // 01 = Add, 10 = Subtract, 00 = Idle
    output reg signed [3:0] data_out // Current memory value or ALU result
);

    // 64-bit SRAM (16 words x 4 bits)
    reg signed [3:0] sram [15:0];
    
    // Internal address counter to track where we are in memory
    reg [3:0] addr_counter;
    
    // Internal registers for ALU synchronization
    reg [3:0] reg_a, reg_b;

    always @(posedge clk or posedge reset) begin
        if (reset) begin
            addr_counter <= 4'b0000;
            data_out     <= 4'b0000;
            reg_a        <= 4'b0000;
            reg_b        <= 4'b0000;
        end 
        else begin
            // --- STEP 1: Automatic Memory Loading ---
            // Each time load_en is pulsed, data is saved and counter moves up
            if (load_en) begin
                sram[addr_counter] <= data_in;
                data_out           <= data_in; // Show data for verification
                addr_counter       <= addr_counter + 1;
            end

            // --- STEP 2: ALU Operation ---
            // Using the data already stored in the previous memory slots
            if (op_sel == 2'b01 || op_sel == 2'b10) begin
                reg_a <= sram[addr_counter - 2]; // Get first number
                reg_b <= sram[addr_counter - 1]; // Get second number

                case (op_sel)
                    2'b01: begin
                        data_out <= reg_a + reg_b;
                        sram[addr_counter] <= reg_a + reg_b; // Store result
                    end
                    2'b10: begin
                        data_out <= reg_a - reg_b;
                        sram[addr_counter] <= reg_a - reg_b; // Store result
                    end
                endcase
                
                // Increment counter after storing result to be ready for next operation
                addr_counter <= addr_counter + 1;
            end
        end
    end
endmodule
